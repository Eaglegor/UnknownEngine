111  void MessageDispatcher::deliverMessage ( const PackedMessage &msg, IMessageDeliveryPolicy* delivery_policy ) const
112  {
113  const MessageListenersList* list = getRegisteredListeners ( msg.getMessageTypeId () );
114  if ( list != nullptr )
115  {
116  for ( RegisteredListener registered_listener : *list )
117  {
118  if ( delivery_policy == nullptr || delivery_policy->allowDeliveryToListener ( registered_listener.listener ) )
119  {
120  if ( registered_listener.receive_policy == nullptr || registered_listener.receive_policy->acceptMessage ( msg ) )
121  {
122  registered_listener.listener->processMessage ( msg );
123  if( delivery_policy != nullptr )
124  {
125  delivery_policy->notifySuccessulyDelivered();
126  }
127  }
128  }
129  }
130  }
131  if( delivery_policy != nullptr )
132  {
133  delivery_policy->notifyDeliveryFinished();
134  }
